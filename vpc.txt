绿化.bat ==>

echo off
COPY VMM.SYS %SystemRoot%\SYSTEM32\drivers\VMM.SYS /Y
::regsvr32 /s msxml6.dll
regsvr32 /s msxml4.dll
regedit /s vmm.reg
cd .\Utility
.\install.bat
echo on
